        src/DoAllWrap.cpp
        src/ParaMeter.cpp
        src/Queue.cpp
)

  ${CMAKE_SOURCE_DIR}/libllvm/include
  ${CMAKE_SOURCE_DIR}/libsubstrate/include
  ${CMAKE_SOURCE_DIR}/libruntime/include
